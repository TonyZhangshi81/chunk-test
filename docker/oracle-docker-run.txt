
# oracle23 镜像：

docker run -d `
--name oracle23ai `
-p 1521:1521 -p 5500:5500 `
-e ORACLE_PWD=YourPassword123 `
-e ORACLE_CHARACTERSET=AL32UTF8 `
-v C:\docker\oradata:/opt/oracle/oradata `
container-registry.oracle.com/database/free:latest

# 注意：

首次启动等待时间：Oracle 23ai 镜像在首次启动时需要进行初始化，整个过程可能需要 2-5 分钟。
可以使用 docker logs oracle23ai -f 命令来实时查看启动日志，直到看到 DATABASE IS READY TO USE! 的字样

# 创建本地用户

在 PDB（可插拔数据库） 中可以直接创建用户名不带前缀的普通用户（本地用户）

-- 1. 查看当前有哪些 PDB（先确认 PDB 名称）
SELECT name, open_mode FROM v$pdbs;

-- 2. 切换到 PDB（通常默认叫 FREEPDB1 或 XEPDB1）
ALTER SESSION SET CONTAINER = FREEPDB1;

-- 3. 在 PDB 中创建普通用户
CREATE USER tonyzhangshi IDENTIFIED BY Tony19811031
DEFAULT TABLESPACE USERS
QUOTA UNLIMITED ON USERS;

-- 4. 授予基本权限
GRANT CONNECT, RESOURCE, CREATE SESSION TO tonyzhangshi;


# 在当前用户下创建表
eg:
-- 直接创建表即可，会自动存放在 tonyzhangshi 的模式下
CREATE TABLE my_table (
    id NUMBER PRIMARY KEY,
    name VARCHAR2(100),
    created_date DATE
);
